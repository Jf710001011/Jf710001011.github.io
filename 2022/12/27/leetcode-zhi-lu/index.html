2023-12-16 287.寻找重复数

我思

  1. 联想：异或
  2. 排序？不符合要求

实现

  1. 规律，cnt[i]表示nums中小于i的元素个数，主要重心在于以target(重复元素值)的cnt值为中轴，观察两端cnt值规律

  2. Code

    class Solution {
   public:
    int findDuplicate(vector<int>& nums) {
        // 考虑三种重复情况，实际上只要考虑两种
        int left = 1;
        int right = nums.size() - 1;
        int ans;
        while (left <= right) {
            // 二分查找(下标有序，含义为<=i的元素个数)
            int mid = (left + right) >> 1;
            int count = 0;  // <=mid的元素个数

            for (int i = 0; i < nums.size(); i++) {
                if (nums[i] <= mid) {
                    count++;
                }
            }

            // ans 只能通过mid确定
            if (count <= mid) {  // 得在mid 右边查找
                // 此时的mid 一定不是ans
                left = mid + 1;
            } else {
                // 此时的mid有可能是ans,但是还要继续使用left和right缩小范围
                // 因此ans 只能在此处赋值
                right = mid - 1;
                ans = mid;
            }
        }
        return ans;
    }
};

2023-12-05 279. 完全平方数

我思

  1. 找到与该数字x轴左侧最接近的平方数，从大到小试探，保证最数字数量最少

实现

  1. dp

    1. 依据题目的要求写出状态表达式：f[i] 表示最少需要多少个数的平方来表示整数 i。

      这些数必然落在区间 \[[1,\sqrt{n}]\]。我们可以枚举这些数，假设当前枚举到 j，那么我们还需要取若干数的平方，构成 此时我们发现该子问题和原问题类似，只是规模变小了。这符合了动态规划的要求，于是我们可以写出状态转移方程。

  2. 寻找问题的特殊性：如1的平方是1，则一个非常大的数，可以有好多个1相加而成

  3. 该提出的疑问：

    1. 是否所有数都必定符合由平方和
    2. 在满足1的情况下，这种组合有多少种，有没有一个范围

  4. Code

    class Solution {
   public:
    int numSquares(int n) {
        vector<int> dp(n + 1);  // 边界条件dp[0] = 0
        for (int i = 1; i <= n; i++) {
            dp[i] = i;  // 全是平方为1的和
            for (int j = 1; j * j <= i; j++) {
                // 每一轮二次循环都要会更新dp[i]，因为每个数字有多种组合
                dp[i] = min(dp[i], dp[i - j * j] + 1);  // 转移方程
            }
        }
        return dp[n];
    }
};

    class Solution {
   public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        // 而如果我们遍历的元素一直在遍历范围的右上角（x,y）
        // 遍历的范围局限于以(x，y)为右上角的矩阵中
        // 那么此刻我们保证了该元素是该范围中行最小，列最大的元素
        int rows = matrix.size();
        int cols = matrix[0].size();
        int i = 0;
        int j = cols - 1;

        while (i < rows && j >= 0) {
            if (matrix[i][j] > target) {  // 行中已是最大，只能列减小
                j--;
            } else if (matrix[i][j] == target) {
                return true;
            } else {  // 列中已是最小，只能行增加
                i++;
            }
        }
        return false;
    }
};

2023-11-29 238. 除自身以外数组的乘积

我思

  1. 除法是最容易想到的，但是 0时失效

实现

  1. 另一个角度是：该元素**左边所有的乘积*右边所有的乘积**

  2. 时间的优化必然是牺牲空间，那会牺牲在哪里呢

  3. Code

    class Solution {
   public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int len = nums.size();
        vector<int> ans(len);

        // 记录每个元素的左侧乘积和右侧乘积
        ans[0] = 1;
        for (int i = 1; i < len; i++) {
            ans[i] = ans[i - 1] * nums[i - 1];
        }

        // 题目设定：输出数组的空间不考虑
        // 所以将一侧的乘积数组作为输出数组
        // 而另一侧则只需要记住该元素相关的乘积，而不是乘积数组，因此只需要一个变量记录
        int right = 1;  // 该值右侧所有元素的乘积
        for (int i = len - 1; i >= 0; i--) {
            ans[i] = ans[i] * right;
            right *= nums[i];  // 向左更新right
        }
        return ans;
    }
};

      class Solution {
   public:
    int maximalSquare(vector<vector<char>>& matrix) {
        int row = matrix.size();
        int col = matrix[0].size();

        if (row == 0 || col == 0) {
            return 0;
        }

        int ans = 0;
        vector<vector<int>> dp(row, vector<int>(col));
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (matrix[i][j] == '1') {  // 注意是char型
                    if (i == 0 || j == 0) {
                        dp[i][j] = 1;
                    } else {
                        dp[i][j] = min(min(dp[i - 1][j], dp[i - 1][j - 1]),
                                       dp[i][j - 1]) +
                                   1;
                    }
                }
                ans = max(ans, dp[i][j]);
            }
        }
        int maxSquare = ans * ans;
        return ans * ans;
    }
};

void quickSort(std::vector<int>& nums, int left, int right) {
    if (left < right) {
        srand(time(nullptr));
        int random = rand() % (right - left + 1) + left;
        int pivot = nums[random];

        std::swap(nums[left], nums[random]);

        int j = left;
        for (int i = left + 1; i < right; i++) {
            if (nums[i] >= pivot) {
                std::swap(nums[j + 1], nums[i]);
                j++;
            }
        }

        std::swap(nums[j], nums[left]);
        quickSort(nums, left, j);
        quickSort(nums, j + 1, right);
    }
}

      #include <algorithm>
#include <iostream>
#include <vector>

using namespace std;

//  调整堆
void heapify(vector<int>& nums, int n, int i) {
    int indexLargest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    // 如果左子节点大于根节点
    if (left < n && nums[left] > nums[indexLargest]) {
        indexLargest = left;
    }

    // 如果右子节点大于根节点
    if (right < n && nums[right] > nums[indexLargest]) {
        indexLargest = right;
    }

    // 调整以i为根节点的树
    if (i != indexLargest) {  // 最大节点不是根节点
        swap(nums[i], nums[indexLargest]);
        // indexLargest：是与根节点产生交换的节点下标，以此为根节点的子树也可能需要调整
        heapify(nums, n, indexLargest);
    }
}

// 堆排序
void heapSort(vector<int>& nums) {
    int n = nums.size();

    // 自底向上调整：构建大根堆,从最后一个非叶子节点开始
    for (int i = n / 2 - 1; i >= 0; i--) {
        heapify(nums, n, i);
    }

    // 删除堆顶元素并向下重新调整
    for (int i = n - 1; i > 0; i--) {
        swap(nums[0], nums[i]);
        // 从根节点开始调整
        heapify(nums, i, 0);
    }
}

      class Solution {
   public:
    void quickSelectKth(vector<int>& nums, int left, int right, int k) {
        srand(time(nullptr));
        int pivot = rand() % (right - left + 1) + left;
        swap(nums[left], nums[pivot]);

        int j = left;

        for (int i = left + 1; i <= right; i++) {
            if (nums[i] >= nums[left]) {
                swap(nums[j+1], nums[i]);
                // j++ 之后的j = 上一行的j+1;出循环后保证了j是>=nums[pivot]的最小值
                j++;
            }
        }

        // 交换后，保证了nums[j]是在0-j序列中的最小值，因此如果k<j,则nums[k]>nums[j]
        // 因此往左侧搜寻
        swap(nums[j], nums[left]);
        // while循环结束，i=j,此时的位置就是pivot排序的最终位置，也就确认了pivot是第几大

        // 如果k=j,则已经达到目的
        if (k < j) {
            quickSelectKth(nums, left, j - 1, k);
        } else if (k > j) {
            quickSelectKth(nums, j + 1, right, k);
        }
    }

    int findKthLargest(vector<int>& nums, int k) {
        // 注意第k大的数在数组中排序是k-1
        quickSelectKth(nums, 0, nums.size() - 1, k - 1);
        return nums[k-1];
    }
};

    class Solution {
   public:
    int quickSearchKth(vector<int>& nums, int k) {
        srand(time(nullptr));
        int pivot = nums[rand() % nums.size()];

        vector<int> big, equal, small;
        for (int num : nums) {
            if (num > pivot) {
                big.push_back(num);
            } else if (num < pivot) {
                small.push_back(num);
            } else {
                equal.push_back(num);
            }
        }

        if (k <= big.size()) {
            return quickSearchKth(big, k);
        } else if (nums.size() - small.size() < k) {
            // 在全局中是第k大,而在samll中不是
            return quickSearchKth(small, k - big.size() - equal.size());
        }

        return pivot;
    }

    int findKthLargest(vector<int>& nums, int k) {
        return quickSearchKth(nums, k);
    }
};

2023-11-15 200.岛屿数量

我思

  1. 理解1座岛屿的概念：
    1. 岛屿总是被水包围，
    2. 并且每座岛屿只能由水平方向和/或竖直方向上相邻的陆地连接形成
  2. 识别到1 （i,j），那么就检测它的周围（[i-1,j],[i+1,j],[i,j-1],[i,j+1]；注意：考虑到边界原因，这些坐标并不一定都存在）是否为1，
    1. 如果是1，那么就是同一座岛屿，并把它标记为已处理
      1. 1座岛屿面积确定的标志：所有1的上下左右均为0，否则加入新的1继续延伸
    2. 如果为0，那么就是岛屿的一处边界

实现

  1. Code

    class Solution {
   public:
    // dfs
    // 进行dfs的次数便是岛屿的个数
    void dfs(vector<vector<char>>& grid, int r, int c) {
        // 遍历到的1复位为0，避免重复处理
        grid[r][c] = '0';

        // dfs 这个“1”的四周
        int rc = grid.size();
        int cc = grid[0].size();

        if (r - 1 >= 0 && grid[r - 1][c] == '1')
            dfs(grid, r - 1, c);
        if (r + 1 < rc && grid[r + 1][c] == '1')
            dfs(grid, r + 1, c);
        if (c - 1 >= 0 && grid[r][c - 1] == '1')
            dfs(grid, r, c - 1);
        if (c + 1 < cc && grid[r][c + 1] == '1')
            dfs(grid, r, c + 1);
    }

    int numIslands(vector<vector<char>>& grid) {
        int rc = grid.size();
        if (!rc)
            return 0;
        int cc = grid[0].size();

        int nums_islands = 0;
        for (int r = 0; r < rc; r++) {
            for (int c = 0; c < cc; c++) {
                if(grid[r][c] == '1') {
                    nums_islands++;
                    dfs(grid,r,c);
                }
            }
        }
        return nums_islands;
    }
};

2023-11-19: 207. 课程表

我思

  1. 示例是不是不够，应该还有很多情况没有列出来吧（但感觉也不太可能）

实现

  1. 每个课程表示为图中的一个顶点及其每门课程之间的关系构成了一个图

  2. 给定一个有向无环图，拓扑排序是将图中的顶点按顺序排列，使得对于图中任意一条有向边，其起点顶点在其终点顶点之前。

  3. 如果会互相依赖，实际则表现为成环，则不可能完成课程的学习

  4. 说明：

    1. 对于图中的任意一个节点，它在搜索的过程中有三种状态，即：

      「未搜索」：我们还没有搜索到这个节点；

      「搜索中」：我们搜索过这个节点，但还没有回溯到该节点，即该节点还没有入栈，还有相邻的节点没有搜索完成）；

      「已完成」：我们搜索过并且回溯过这个节点，即该节点已经入栈，并且所有该节点的相邻节点都出现在栈的更底部的位置，满足拓扑排序的要求。

  5. Code

    class Solution {
   private:
    vector<vector<int>> edges;
    vector<int> visit;
    bool finished = true;

   public:
    void dfs(int u) {
        // 标记为已被访问
        visit[u] = 1;

        // 访问u的邻接点
        for (int v : edges[u]) {
            // 有环（即不能完成所有课程）
            if (!finished) {
                return;
            }

            if (visit[v] == 0) {
                // 如果该邻接点未被访问过，则继续深度优先搜索
                dfs(v);
            } else if (visit[v] == 1) {
                // 如果该节点之前便处于搜索中，那么说明存在环
                finished = false;
                return;
            }
        }
        // 返回u,u及其邻接点搜索完成	
        visit[u] = 2;
    }
    bool canFinish(int numCourses, vector<vector<int>>& prerequisites) {
        edges.resize(numCourses);
        visit.resize(numCourses);

        for (const auto& info : prerequisites) {
            edges[info[1]].push_back(info[0]);
        }

        for (int i = 0; i < numCourses && finished; ++i) {
            if (visit[i] == 0) {
                dfs(i);
            }
        }
        return finished;
    }
};

    class Solution {
   public:
    int minDistance(string word1, string word2) {
        // 动态规划
        // 定义 dp[i][j] : dp[i][j] 代表 word1 中前 i 个字符，变换到 word2 中前
        // j 个字符，最短需要操作的次数:需要考虑 word1 或 word2
        // 状态转移 :
        // 增，dp[i][j] = dp[i][j - 1] + 1
        // 删，dp[i][j] = dp[i - 1][j] + 1
        // 改，dp[i][j] = dp[i - 1][j - 1] + 1
        // 按顺序计算，当计算 dp[i][j] 时，dp[i - 1][j] ， dp[i][j - 1] ， dp[i
        // - 1][j - 1] 均已经确定了 配合增删改这三种操作，需要对应的 dp
        // 把操作次数加一，取三种的最小 如果刚好这两个字母相同 word1[i - 1] =

        // vector 的初始化方法为 vector(size,value)

        for (int i = 0; i < dp.size(); i++) {
            dp[i][0] = i;
        }
        for (int j = 0; j < dp[0].size(); j++) {
            dp[0][j] = j;
        }

        for (int i = 1; i < dp.size(); i++) {
            for (int j = 1; j < dp[0].size(); j++) {

                // ，操作不用加一
                dp[i][j] =
                    min(dp[i][j - 1], min(dp[i - 1][j], dp[i - 1][j - 1])) + 1;

                // 放在上句的后面是因为当两个单词当前字母相等时，放在后面才能确保dp[i][j]的正确性
                if (word1[i - 1] == word2[j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1];
                }
            }
        }
        // 最后一个行中的最后一个元素。
        return dp.back().back();
    }
};

// 但如果是链表，则双向可以更快确定插入位置/删除位置的前后节点
struct DLinkedNode {
    int key, value;
    DLinkedNode* prev;
    DLinkedNode* next;
    DLinkedNode() : key(0), value(0), prev(nullptr), next(nullptr) {}
    DLinkedNode(int _key, int _value)
        : key(_key), value(_value), prev(nullptr), next(nullptr) {}
};

class LRUCache {
   private:
    unordered_map<int, DLinkedNode*> cache;
    DLinkedNode* head;
    // 移除尾部节点（被置换的节点）
    DLinkedNode* tail;
    // 比较size和capacity确定是否需要置换
    int size;
    int capacity;

   public:
    LRUCache(int _capacity) : capacity(_capacity), size(0) {
        head = new DLinkedNode();
        tail = new DLinkedNode();
        head->next = tail;
        tail->prev = head;
    }

    int get(int key) {
        if (!cache.count(key)) {
            return -1;
        }

        DLinkedNode* node = cache[key];
        // 刷到最新
        moveToHead(node);
        return node->value;
    }

    void put(int key, int value) {
        if (!cache.count(key)) {
            DLinkedNode* node = new DLinkedNode(key, value);
            cache[key] = node;
            // 刷新
            addToHead(node);
            ++size;

            // 超出容量
            if (size > capacity) {
                DLinkedNode* removed = removeTail();
                cache.erase(removed->key);
                delete removed;
                --size;
            }
        } else {
            DLinkedNode* node = cache[key];
            node->value = value;
            moveToHead(node);
        }
    }

    void addToHead(DLinkedNode* node) {
        node->prev = head;
        node->next = head->next;
        head->next = node;
        node->next->prev = node;
    }

    void removeNode(DLinkedNode* node) {
        node->prev->next = node->next;
        node->next->prev = node->prev;
    }

    void moveToHead(DLinkedNode* node) {
        removeNode(node);
        addToHead(node);
    }

    DLinkedNode* removeTail() {
        DLinkedNode* node = tail->prev;
        removeNode(node);
        return node;
    }
};

    class Solution {
   public:
    ListNode* detectCycle(ListNode* head) {
        // 从头节点出发：
        // 快指针走2步，慢指针走一步。
        // 重合需要满足的要求：快指针追上慢指针（在环内），快指针刚好超过慢指针n圈环
        ListNode* fast = head;
        ListNode* slow = head;

        while (true) {
            if (!fast || !(fast->next)) {
                return nullptr;
            }

            fast = fast->next->next;
            slow = slow->next;
            if (fast == slow) {
                // f（快指针步数）；s(慢指针步数)；b(环的步数)
                // 在第一次相遇时满足：
                // 1. f = 2s; (2倍速度)
                // 2. f = s+nb
                // 由以上两式知s=nb
                // 画个示意图：假设快慢指针在环内的同一点即s+(n-1)b处，则再次相遇快指针要超过慢指针一圈
                break;
            }
        }
        // a(链表头结点到环起点的步数)；x（从环起点到当前位置的步数）;k<n(绕环的圈数)
        // 1. s = nb
        // 2. s = （n-1）b+a+x
        // 因此慢指针在此时的位置（第一次相遇）再走a步必定在环的起始处，
        // 如何确定走a步：让另一指针在头节点出发，该指针与慢指针相遇则确定是走了a步
        fast = head;
        while (slow != fast) {
            fast = fast->next;
            slow = slow->next;
        }
        return slow;
    }
};

        unordered_set<int> nums_set;
        for (int num : nums) {
            nums_set.insert(num);
        }

        int longestSequence = 0;

        for (int num : nums) {

            }
        }

    class Solution {
   public:
    bool exist(vector<vector<char>>& board, string word) {
        int m = board.size(), n = board[0].size();
        // 记录是否访问过
        vector<vector<bool>> visited(m, vector<bool>(n, false));
        // 遍历每个点
        for (int i = 0; i < m; ++i)
            for (int j = 0; j < n; ++j) {
                // 回溯
                if (backTracking(board, visited, word, 0, i, j))
                    return true;
            }
        return false;
    }

   private:
    // 回溯
    bool backTracking(vector<vector<char>>& board,
                      vector<vector<bool>>& visited,
                      const string& word,
                      int strCur,
                      int i,
                      int j) {
        if (strCur == word.size())
            return true;
        // 越界、已访问(形成环状)、不匹配
        if (i < 0 || i >= board.size() || j < 0 || j >= board[0].size() || visited[i][j] ||
            board[i][j] != word[strCur])
            return false;
        visited[i][j] = true;
        if (backTracking(board, visited, word, strCur + 1, i - 1, j) ||
            backTracking(board, visited, word, strCur + 1, i + 1, j) ||
            backTracking(board, visited, word, strCur + 1, i, j - 1) ||
            backTracking(board, visited, word, strCur + 1, i, j + 1))
            return true;
        visited[i][j] = false;
        return false;
    }
};

   public:
    int minPathSum(vector<vector<int>>& grid) {
        // dp；每次只能向右或者向下
        // 误区：每次选最小路径去走，最终到右下角不一定是最小路径
        // 但是此位置的最小路径
        // 前一步一定是从上一步的更小路径走出来的，因此我们仍然是计算每个格子的最小

        for (int i = 1; i < row; ++i)
            for (int j = 1; j < column; ++j)
                dp[i][j] = min(dp[i - 1][j], dp[i][j - 1]) + grid[i][j];

    class Solution {
   public:
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        // 结果集
        vector<vector<int>> ans;
        // 边界处理
        if (intervals.size() == 0) {
            return ans;
        }

        // 依据子区间左值排序
        sort(intervals.begin(), intervals.end());
        int left = intervals[0][0], right = intervals[0][1];
        for (int i = 1; i < intervals.size(); i++) {
            // 如果当前区间的左值小于等于上一个区间的右值，说明有重叠
            if (intervals[i][0] <= right) {
                // 更新右值
                right = max(right, intervals[i][1]);
            } else {
                // 没有重叠，将上一个区间加入结果集
                ans.push_back({left, right});
                // 更新左右值
                left = intervals[i][0];
                right = intervals[i][1];
            }
        }
        // 最后一个子区间处理后需要加入结果集
        ans.push_back({left, right});
        return ans;
    }
};

            return accumulate(arr.begin(), arr.end(), 0u, [&](size_t acc, int num) {
                return (acc << 1) ^ fn(num);
            });
        };

        unordered_map<array<int, 26>, vector<string>, decltype(arrayHash)> mp(0, arrayHash);
        for (string& str: strs) {
            array<int, 26> counts{};
            int length = str.length();
            for (int i = 0; i < length; ++i) {
                counts[str[i] - 'a'] ++;
            }
            mp[counts].emplace_back(str);
        }
        vector<vector<string>> ans;
        for (auto it = mp.begin(); it != mp.end(); ++it) {
            ans.emplace_back(it->second);
        }
        return ans;
    }
};

class Solution {
   public:
    int search(vector<int>& nums, int target) {
        int left = 0;
        int right = nums.size() - 1;

        // 总的来说比起常规
        while (left < right) {
            int mid = (left + right) >> 1;
            if (nums[mid] == target) {
                return mid;
            }

            if (nums[left] <= nums[mid]) {    // 前半段有序
                if (target >= nums[left] && target < nums[mid]) {    // 先查找前半段;注意等号
                    right = mid - 1;
                } else {    // 在后半段查找
                    left = mid + 1;
                }
            } else {    // 后半段有序
                if(target > nums[mid] && target <= nums[right]) {    // 先查找后半段
                    left = mid + 1;
                } else {    // 在前半段查找
                    right = mid - 1;
                }

            }
        }
        return nums[left] == target ? left : -1;
    }
};

class Solution {
   public:
    void nextPermutation(vector<int>& nums) {
        // 双指针遍历
        for (int left = nums.size() - 2; left >= 0; left--) {
            for (int right = nums.size() - 1; right > left; right--) {    // 是否考虑数组只有一个数字
                if (nums[left] < nums[right]) {
                    swap(nums[left], nums[right]);
                    sort(nums.begin() + left + 1, nums.end());
                    return;
                }
            }
        }
        reverse(nums.begin(), nums.end());
    }
};

      class Solution {
   public:
    vector<string> generateParenthesis(int n) {
        vector<string> ans;
        string path;
        backTrace(ans, path, 0, 0, n);    // path 直接放空也行
        return ans;
    }

    // ans 得是引用
    // left 是已用还是剩余需明确
    void backTrace(vector<string>& ans, string path, int left, int right, int n) {
        if (left > n || right > n || right > left) {    // 递归终止条件
            return;
        }

        if (left == n && right == n) {    // 压入返回值的条件(全部括号得用完，且左右相等)
            ans.push_back(path);
            return;     // 压入之后返回上一层 
        }

        // 路径的生成在传参时完成
        // path 参数需要特别注意不能用引用，且使用加法更不会出错，因为返回上一层时，path的值还得是上一层的值，而不能是下一层的值
        backTrace(ans, path + "(", left + 1, right, n);
        backTrace(ans, path + ")", left, right + 1, n);
    }
};

class Solution {
   private:
    const string s[10] = {
        "", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz",
    };

   public:
    vector<string> ans;
    string path;
    void backtracking(string digits, int level) {
        if (level == digits.size()) {    // 终止条件：纵向遍历完(每条分支)
            ans.push_back(path);
            return; // 返回上一层
        }

        int digit = digits[level] - '0';
        for (int i = 0; i < s[digit].size(); ++i) {
            path.push_back(s[digit][i]);
            // backtracking(digits, ++level);   // ++level的执行时间(所在层数)，这是错误的
            backtracking(digits, level+1);     // 返回上一层后，level的值还是本层的值，不会加一
            path.pop_back();    // 回溯，弹出最后一层
        }
    }

    vector<string> letterCombinations(string digits) {
        if (digits.size() == 0) {
            return ans;
        }

        backtracking(digits,0);
        return ans;
    }
};

class Solution {
   public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        sort(nums.begin(), nums.end());    // 先排序
        vector<vector<int>> ans;
        int left = 0;
        int mid;
        int right = nums.size() - 1;
        if (nums[left] > 0 || nums[right] < 0) {
            return ans;
        }

        // 每次固定左指针，最中、右指针处理
        for (left; left < nums.size() - 2; left++) {
            if(left != 0 && nums[left] == nums[left-1]) {   // left 也得去重
                continue;   // 不是break
            }

            // left 移动，则mid,right重置
            mid = left + 1;
            right = nums.size() - 1;

            while (mid < right) {
                if (nums[left] + nums[mid] + nums[right] > 0) {
                    right--;
                } else if (nums[right] + nums[mid] + nums[left] < 0) {
                    mid++;
                } else {
                    ans.push_back(vector<int>{nums[left], nums[mid], nums[right]});

                    //  去重
                    // mid 先去重
                    mid++;
                    while (mid < right && nums[mid] == nums[mid - 1]) {
                        mid++;
                    }

                    right--;
                    while (mid < right && nums[right] == nums[right + 1]) {
                        right--;
                    }

                }
            }
        }
        return ans;
    }
};

11. 盛最多水的容器

  • 我的思路
    • 分析：容积以两轴中短轴为准
      • 两根轴要尽可能远，且短轴尽可能高，能保证最大容积。但是显然最优解不一定要同时满足这两个条件。
      • 这个问题本质是确定两个边界
      • 问题是：如何确定这两个边界呢，暴力解法O(n²)不考虑
      • 两个边界很容易联想到双指针
      • 双指针的移动规则怎么确定呢，是否要遍历每一种情况呢（显然不要，不然不就是暴力吗）
  • 参考答案
    • 两个边界肯定不是同时移动，那么就是一次移动一个
    • 每次不管左移还是右移，底边都是-1，因此得确保高尽量高，也就是长轴保持，移动短轴寻找更长轴
    • 不容易注意到的一个问题是：如果要保证之后的结果比现在更大，那么最短轴一定要更长（因为底边再减小），也就是说每次移动，最短轴一定要变化。如果最短轴不变，那么这次移动是无效的。
  • submit
class Solution {
   public:
    int maxArea(vector<int>& height) {
        int left = 0;
        int right = height.size() - 1;
        int ans = 0;
        while (left < right) {
            ans = max(ans, (right - left) * min(height[left], height[right]));
            if (height[left] < height[right]) {
                left++;
            } else {
                right--;
            }
        }
        return ans;
    }
};

class Solution {
   public:
    int minimumLength(string s) {    // 双指针法
        if (s.length() <= 1) {
            return s.length();
        }
        int left = 0;
        int right = s.length() - 1;
        while (left < right && s[left] == s[right]) {
                left++;
                while (left < right && s[left - 1] == s[left]) {    // 删除左边的全部
                    left++;
                }
                right--;
                while (left < right && s[right + 1] == s[right]) {    // 删除右边的全部
                    right--;
                }
            }
            return right - left + 1;
        }
};

class Solution {
   public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        for (int i = 0; i < nums.size(); i++) {
            nums[abs(nums[i]) - 1] = -abs(nums[abs(nums[i]) - 1]); // 取绝对值的原因在于有些数字会重复出现，第一次出现就会更改为负数，而负数不能作为下标
        }
        vector<int> result;
        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] > 0) {
                result.push_back(i + 1);    // 这里是i+1
            }
        }
        return result;
    }
};
