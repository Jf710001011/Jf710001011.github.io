ANSWER: 

SELECT DIRECTOR,SUM(DOMESTIC_SALES+INTERNATIONAL_SALES),COUNT(ID),AVG(DOMESTIC_SALES+INTERNATIONAL_SALES) AS AVERAGE
FROM movies
JOIN Boxoffice
ON ID = MOVIE_ID
GROUP BY DIRECTOR
HAVING COUNT(ID)>1
ORDER BY AVERAGE DESC
LIMIT 1;
