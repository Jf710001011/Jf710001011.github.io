  5. Analogy:

    • Think of the JDK as a complete chef's kitchen with all the tools and ingredients needed to cook (develop) a delicious meal (Java application).
    • The JRE is like a pre-packaged microwave meal - you just need the microwave (JVM) to heat it up (run) and enjoy it.
    • The JVM is like the heating element inside the microwave, responsible for the actual cooking process, regardless of the meal type.

    Key Takeaways:

    • JDK contains JRE and JVM, while JRE only contains JVM.
    • JDK is for development, JRE is for execution.
    • JVM is the platform-independent execution engine

  2. arr = arr2

    int[] arr1 = {1, 2, 3};
int[] arr2 = arr1;

    In this case, arr1 and arr2 both point to the same array in memory. If you change an element in arr1, the change will be reflected in arr2 as well, because they're both pointing to the same data.

    If you want to create a copy of an array, you need to create a new array and copy each element individually. You can do this manually with a for loop, or you can use the System.arraycopy() method or Arrays.copyOf() method.

    import java.util.Random;

public class Role {
    private String name;
    private int bloodVol;
    private char gender;
    private String face;

    public String getName() {
        return name;
    }

    // face
    String[] boyFaces = { "风流俊雅", "气宇轩昂", "相貌英俊", "五官端正", "相貌平平", "一塌糊涂", "面目狰狞" };
    String[] girlFaces = { "美奂绝伦", "沉鱼落雁", "亭亭玉立", "身材娇好", "相貌平平", "相貌简陋", "惨不忍睹" };

    // kungfu
    String[] attackMoves = {
            "%s使出了一招【背心钉】，转到对方身后，一掌向%s背心的灵台穴拍去。",
            "%s使出了一招【a】，转到对方身后，一掌向%s的[b]拍去。",
            "%s使出了一招【c】，转到对方身后，一掌向%s的[d]拍去。",
    };
    String[] injuryActions = {
            "结果%s退了半步，毫发无损",
            "结果%s退了1步，毫发无损",
            "结果%s退了2步，毫发无损",
    };

    public Role() {

    }

    public Role(String name, int bloodVol, char gender, String face) {
        this.name = name;
        this.bloodVol = bloodVol;
        this.gender = gender;
        setFace();
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getBloodVol() {
        return bloodVol;
    }

    public void setBloodVol(int bloodVol) {
        this.bloodVol = bloodVol;
    }

    public char getGender() {
        return gender;
    }

    public void setGender(char gender) {
        this.gender = gender;
    }

    public String getFace() {
        return face;
    }

    public void setFace() {
        // Random looks
        Random rand = new Random();
        if (this.gender == '男') {
            int index = rand.nextInt(boyFaces.length);
            this.face = boyFaces[index];
        } else {
            int index = rand.nextInt(girlFaces.length);
            this.face = girlFaces[index];
        }
    }

    // display role inforamtion
    public void showRoleInfo() {
        System.out.println("name: " + this.name);
        System.out.println("bloodVolume: " + this.bloodVol);
        System.out.println("gender: " + this.gender);
        System.out.println("face: " + this.face);
    }

    // attack
    public void attack(Role role) {
        Random rand = new Random();
        int index = rand.nextInt(attackMoves.length);

        System.out.printf(attackMoves[index], this.getName(), role.getName());
        System.out.println();

        // Randomly reduce blood volume
        int reduction = rand.nextInt(20) + 1;
        role.setBloodVol(role.getBloodVol() - reduction);
        System.out.println(role.getName() + "剩余血量为：" + role.getBloodVol());

        if (role.getBloodVol() <= 0) {
            System.out.println(this.getName() + "Attack too hard and killed " + role.getName());
        }

        if (role.getBloodVol() > 70) {
            System.out.printf(injuryActions[0], this.getName(), role.getName());
        } else if (role.getBloodVol() > 40) {
            System.out.printf(injuryActions[1], this.getName(), role.getName());
        } else if (role.getBloodVol() > 0) {
            System.out.printf(injuryActions[2], this.getName(), role.getName());
        }
        System.out.println();
    }
}

    package StringCompare;

public class StringCompare {
    public static void main(String[] args) {
        String s1 = "abc";
        String s2 = "abc";
        char[] chs = { 'a', 'b', 'c' };
        String s3 = new String(chs);
        // Compare the memory address of th     e object
        System.out.println(s1 == s2); // true
        System.out.println(s1 == s3); // false
        System.out.println("--------");

        // Compare the content of the object
        System.out.println(s1.equals(s2)); // true
        System.out.println(s1.equals(s3)); // true
        System.out.println("--------");

        // NullPointerException
        String s4 = null;
        System.out.println("abc".equals(s4)); // false
        // System.out.println(s4.equals("abc")); // NullPointerException
        System.out.println("--------");

        // Compare the content of the object and ignore the case
        String s5 = "Java";
        String s6 = "java";
        System.out.println(s5.equals(s6)); // false
        System.out.println(s5.equalsIgnoreCase(s6)); // true
        System.out.println("--------");
    }
}

  7. str.charAt(i) vs str[i]

    1. Immutability: str.charAt(i) doesn't modify the original string, while str[i] can potentially modify it if used for assignment.
    2. Readability: str.charAt(i) is often considered more readable, especially for those new to Java, as it clearly conveys the intent of accessing a character within a string.
    3. Potential for Errors: str[i] can lead to NullPointerException if the string is null, while str.charAt(i) throws a more specific IndexOutOfBoundsException for invalid indices.

  10. What is StringBuilder?

  11. A mutable string class, meaning you can modify its contents after creation.

  12. Designed for efficient string manipulation, especially when performing frequent modifications.

  13. It's a better choice than String for repeated string concatenations as it avoids creating multiple String objects in memory.

  14. Common Use Cases:

    1. Dynamic string building, like assembling HTML content or generating log messages.
    2. Reading and processing text files or user input line by line.
    3. Looping and appending to strings within loops.

  15. Be mindful of thread safety if using StringBuilder in multi-threaded environments.

  19. String splicing principle

    package String_Pratice;

public class StringSplicingPrinciple {

    public static void main(String[] args) {

        // If there are no variables involved, strings are added directly. After
        // compilation, the result is the spliced ​​result.
        // Will reuse strings in the string pool (in the constant pool).
        String s1 = "Hello" + "World";
        String s2 = "HelloWorld";
        System.out.println(s1 == s2); // true
        System.out.println("--------");

        // If there are variables involved
        // Each line of concatenated code creates a new string in memory.
        String s3 = "Hello";
        String s4 = s3 + "World";
        System.out.println(s2 == s4); // false
        System.out.println("--------");

        // All content that needs to be spliced will be put into StringBuilder,
        // which will not create a lot of useless space and save memory.
        StringBuilder sb = new StringBuilder();
        sb.append("Hello");
        sb.append("World");
        String s5 = sb.toString();
        System.out.println(s5 == s2); // false
        System.out.println("--------");
    }

}

  20. StringBuilder Expansion

    package String_Pratice;

public class StringBuilderExpansion {
    public static void main(String[] args) {
        // StringBuilder expansion

        // If the length of the added content is less than 16, it will be stored
        // directly.

        StringBuilder sb = new StringBuilder();

        // By default, a byte array of length 16 is created
        System.out.println(sb.capacity()); // 16

        // If the added content is greater than 16, the capacity will be expanded
        // (original capacity * 2 + 2)
        sb.append("HelloWorldHello");
        System.out.println("capacity: " + sb.capacity()); // 16
        System.out.println("length: " + sb.length()); // 15

        sb.append("HelloWorldHello");
        System.out.println("capacity: " + sb.capacity()); // 34
        System.out.println("length: " + sb.length()); // 15+15

        // If it is not enough after expansion, the actual length shall prevail.
        sb.append("HelloWorldHelloHelloWorldHelloorldHellolo");
        System.out.println("capacity: " + sb.capacity()); // 30+41>(34*2+2)
        System.out.println("length: " + sb.length()); // 30+41

        System.out.println("--------");
    }
}
