class Solution {
   public:
    bool isAnagram(string s, string t) {
        // 哈希表  键与值
        // 一次insert,一次对比
        
        if(s.size() != t.size()) {
            return false;
        }
        
        vector<int> v(26, 0);

        for (char c : s) {
            v[c - 'a']++;
        }

        for (char c : t) {
            v[c - 'a']--;
            if (v[c - 'a'] < 0) {
                return false;
            }
        }

        return true;
    }
};

冒泡排序的优化

//  价格逆序（稳定性）:冒泡排序
void priceSort(vector<book>& v) {
    for (int i = 0; i < v.size() - 1; i++) {
        int flag = 0;
        for (int j = 0; j < v.size() - i - 1; j++) {
            if (v[j].price < v[j + 1].price) {
                swap(v[j], v[j + 1]);
                flag = 1;
            }
        }
        if(!flag) {
            return;
        }
    }
}
